#include "tensorflow/lite/micro/micro_interpreter.h"
#include "tensorflow/lite/micro/micro_mutable_op_resolver.h"
#include "tensorflow/lite/schema/schema_generated.h"
#include "FS.h"
#include "SPIFFS.h"

// 直接从 SPIFFS 读取 .tflite，不需要 .h！
File modelFile;
const char* modelPath = "/papagei_p_dynamic_range_quant.tflite";

const tflite::Model* model = nullptr;
tflite::MicroInterpreter* interpreter = nullptr;
uint8_t* tensor_arena = nullptr;

// 根据你的模型改大小！一般 300000~1000000
constexpr int kTensorArenaSize = 1024 * 600;

void setup() {
  Serial.begin(115200);
  
  // 1. 挂载 SPIFFS
  if(!SPIFFS.begin(true)){
    Serial.println("SPIFFS 挂载失败！");
    return;
  }

  // 2. 直接读取 .tflite 文件
  modelFile = SPIFFS.open(modelPath, "r");
  if (!modelFile) {
    Serial.println("模型文件不存在！");
    return;
  }

  // 3. 读取到内存
  size_t modelSize = modelFile.size();
  uint8_t* modelBuffer = (uint8_t*)malloc(modelSize);
  modelFile.read(modelBuffer, modelSize);
  modelFile.close();

  // 4. 加载 TFLite
  model = tflite::GetModel(modelBuffer);
  
  // 注册需要的层
  static tflite::MicroMutableOpResolver<10> micro_op_resolver;
  micro_op_resolver.AddConv2D();
  micro_op_resolver.AddDepthwiseConv2D();
  micro_op_resolver.AddActivation();
  micro_op_resolver.AddMaxPool2D();
  micro_op_resolver.AddSoftmax();
  micro_op_resolver.AddFullyConnected();
  micro_op_resolver.AddReshape();
  micro_op_resolver.AddAdd();
  micro_op_resolver.AddMul();
  micro_op_resolver.AddMean();

  // 5. 初始化解释器
  tensor_arena = (uint8_t*)malloc(kTensorArenaSize);
  static tflite::MicroInterpreter static_interpreter(
    model, micro_op_resolver, tensor_arena, kTensorArenaSize
  );
  interpreter = &static_interpreter;

  // 分配内存
  TfLiteStatus allocate_status = interpreter->AllocateTensors();
  if (allocate_status != kTfLiteOk) {
    Serial.println("分配内存失败！");
    return;
  }

  Serial.println("✅ 模型加载成功！");
}

void loop() {
  // 你的推理代码
}
